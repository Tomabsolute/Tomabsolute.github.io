Python数据分析基础代码示例

在这篇文章中，我将分享一些Python数据分析的基础代码示例，包括数据导入、清洗、分析和可视化的常用操作。这些代码片段可以作为数据分析项目的起点或参考。

环境准备

首先，我们需要导入必要的库：

# 导入基础数据分析库
import numpy as np
import pandas as pd

# 导入可视化库
import matplotlib.pyplot as plt
import seaborn as sns

# 设置可视化风格
plt.style.use('seaborn-whitegrid')
sns.set_style("whitegrid")

数据导入

从CSV文件导入数据

# 从CSV文件导入数据
df = pd.read_csv('data.csv')

# 查看数据前几行
print(df.head())

# 查看数据基本信息
print(df.info())

# 查看数据统计摘要
print(df.describe())

从Excel文件导入数据

# 从Excel文件导入数据
df_excel = pd.read_excel('data.xlsx', sheet_name='Sheet1')

# 查看数据前几行
print(df_excel.head())

从数据库导入数据

import sqlite3

# 连接到SQLite数据库
conn = sqlite3.connect('database.db')

# 执行SQL查询并将结果加载到DataFrame
df_sql = pd.read_sql_query("SELECT * FROM table_name", conn)

# 关闭连接
conn.close()

# 查看数据
print(df_sql.head())

数据清洗

处理缺失值

# 检查缺失值
print(df.isnull().sum())

# 删除包含缺失值的行
df_cleaned = df.dropna()

# 或者填充缺失值
df_filled = df.fillna({
    'numeric_column': df['numeric_column'].mean(),
    'categorical_column': 'Unknown'
})

# 使用前向填充方法
df_ffill = df.fillna(method='ffill')

# 使用后向填充方法
df_bfill = df.fillna(method='bfill')

处理重复值

# 检查重复行
print(df.duplicated().sum())

# 删除重复行
df_no_duplicates = df.drop_duplicates()

数据类型转换

# 将字符串转换为日期类型
df['date_column'] = pd.to_datetime(df['date_column'])

# 将类别转换为类别类型
df['category_column'] = df['category_column'].astype('category')

# 将字符串转换为数值类型
df['numeric_column'] = pd.to_numeric(df['numeric_column'], errors='coerce')

数据分析

基本统计分析

# 计算相关系数
correlation = df.corr()
print(correlation)

# 分组统计
grouped = df.groupby('category_column')['value_column'].agg(['mean', 'count', 'sum', 'std'])
print(grouped)

# 透视表
pivot = pd.pivot_table(df, 
                       values='value_column', 
                       index='row_category', 
                       columns='column_category', 
                       aggfunc='mean')
print(pivot)

时间序列分析

# 确保日期列是日期时间类型
df['date'] = pd.to_datetime(df['date'])

# 设置日期为索引
df.set_index('date', inplace=True)

# 按月重采样并计算平均值
monthly_mean = df['value'].resample('M').mean()

# 按年重采样并计算总和
yearly_sum = df['value'].resample('Y').sum()

# 计算滚动平均
rolling_mean = df['value'].rolling(window=7).mean()

数据可视化

基础图表

# 折线图
plt.figure(figsize=(10, 6))
plt.plot(df['x'], df['y'], marker='o', linestyle='-')
plt.title('简单折线图')
plt.xlabel('X轴')
plt.ylabel('Y轴')
plt.grid(True)
plt.show()

# 散点图
plt.figure(figsize=(10, 6))
plt.scatter(df['x'], df['y'], alpha=0.6, s=df['size'], c=df['category'], cmap='viridis')
plt.title('散点图')
plt.xlabel('X轴')
plt.ylabel('Y轴')
plt.colorbar(label='类别')
plt.show()

# 柱状图
plt.figure(figsize=(10, 6))
plt.bar(df['category'], df['value'])
plt.title('柱状图')
plt.xlabel('类别')
plt.ylabel('值')
plt.xticks(rotation=45)
plt.show()

使用Seaborn创建高级图表

# 设置图表大小
plt.figure(figsize=(12, 8))

# 热力图
sns.heatmap(correlation, annot=True, cmap='coolwarm', linewidths=0.5)
plt.title('相关系数热力图')
plt.show()

# 成对关系图
sns.pairplot(df[['x', 'y', 'z', 'category']], hue='category')
plt.suptitle('特征之间的成对关系', y=1.02)
plt.show()

# 箱线图
plt.figure(figsize=(12, 6))
sns.boxplot(x='category', y='value', data=df)
plt.title('不同类别的值分布')
plt.show()

# 小提琴图
plt.figure(figsize=(12, 6))
sns.violinplot(x='category', y='value', data=df, inner='quartile')
plt.title('不同类别的值分布密度')
plt.show()

时间序列可视化

# 绘制时间序列数据
plt.figure(figsize=(14, 7))
plt.plot(df.index, df['value'], label='原始数据')
plt.plot(rolling_mean.index, rolling_mean, label='7天滚动平均', linewidth=2)
plt.title('时间序列数据及其滚动平均')
plt.xlabel('日期')
plt.ylabel('值')
plt.legend()
plt.grid(True)
plt.show()

数据导出

# 导出为CSV
df.to_csv('processed_data.csv', index=False)

# 导出为Excel
df.to_excel('processed_data.xlsx', sheet_name='Processed', index=False)

# 导出为JSON
df.to_json('processed_data.json', orient='records')
